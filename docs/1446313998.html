Extracción de imgs relacionadas con el cálculo de un accuracy específico

SELECT y.boxes, y.photo_id, z.image_url
FROM ebisuu_accuracy_newaccuracy as x 
inner join public.ebisuu_accuracy_accuracyphotouserreview as y on x.id = y.accuracy_id 
inner join public.aitrade_core_imagelibrary as z on y.photo_id = z.image_id 
where x.id = 101;
